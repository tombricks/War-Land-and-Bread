RUS_Send_Volunteers_to_Civil_War = {
	reversed = yes

	enable = {
		OR = {
			has_government = Marxism_Leninism
			AND = {
				any_country = {
					civilwar_target = PREV
					has_government = New_Marxism
				}
				NOT = {
					any_country = {
						civilwar_target = PREV
						has_government = Marxism_Leninism
					}
				}
			}
			is_in_array = { RUS.Sphere = THIS }
		}
		has_civil_war = yes
	}

	ai_strategy = {
		type = send_volunteers_desire
		id = RUS
		value = 150
	}
	ai_strategy = {
		type = dont_join_wars_with
		id = RUS
		value = 50
	}
}
